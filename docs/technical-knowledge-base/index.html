Quick Start

1

Prerequisites

  • Node.js 20+ for Trigger.dev tasks
  • Go 1.21+ for runtime engine
  • PostgreSQL 15+ for database
  • Redis 7+ for caching and queues
  • E2B API key for sandbox execution
2

Environment Setup

# Clone the repository
git clone https://github.com/your-org/spiderweb.git
cd spiderweb

# Install dependencies
npm install
go mod download

# Set up environment variables
cp .env.example .env
# Edit .env with your configuration
3

Start Services

# Start the runtime engine
go run cmd/spiderweb/main.go

# Start the dashboard
cd ui/dashboard
npm run dev

# Deploy Trigger.dev tasks
cd infra/trigger
npx trigger deploy
4

Verify Installation

# Check health endpoints
curl http://localhost:13371/health
curl http://localhost:13371/api/v1/health/system

# Access dashboard
open http://localhost:13370

Security Zones

DMZ / Edge Layer

  • API Gateway (13371) - Public REST endpoints
  • WebSocket Gateway (13372) - Real-time communication
  • Dashboard Frontend (13370) - User interface

Application Layer

  • Runtime Engine (13373) - Core system logic
  • Agent Manager (13374) - Agent lifecycle management
  • Channel Handlers (13375-13383) - External integrations
  • Provider Manager (13384) - LLM provider management
  • Skills Registry (13385) - Agent capabilities
  • Event Bus (13386) - Internal messaging

Infrastructure Layer

  • E2B Sandbox (Dynamic) - Agent execution environment
  • OpenClaw Bridge (13387) - External system integration
  • External Services (Various) - Third-party integrations

Data Layer

  • Database (13388) - PostgreSQL primary storage
  • Message Queue (13389) - Redis for caching and queuing
  • File Storage (13390+) - Static file serving

JavaScript Client Example

const ws = new WebSocket('ws://localhost:13372/ws');

ws.onopen = () => {
    // Authenticate
    ws.send(JSON.stringify({
        type: 'auth',
        data: { token: 'your-jwt-token' }
    }));
    
    // Subscribe to events
    ws.send(JSON.stringify({
        type: 'subscribe',
        data: { eventTypes: ['agent_status', 'task_status'] }
    }));
};

ws.onmessage = (event) => {
    const message = JSON.parse(event.data);
    console.log('Received:', message);
    
    if (message.type === 'agent_status') {
        updateAgentStatus(message.data);
    }
};

ws.onclose = () => {
    console.log('WebSocket connection closed');
};

Trigger.dev & E2B Integration

E2B Client Implementation

export class E2BClient {
  private client: E2BClient;

  constructor() {
    this.client = new E2BClient({
      apiKey: process.env.E2B_API_KEY,
      sandboxTemplate: "spiderweb-agent-template"
    });
  }

  async executeAgent(params: {
    agentId: string;
    taskType: string;
    taskPayload: any;
    config: AgentConfig;
  }): Promise {
    const sandbox = await this.client.createSandbox({
      templateId: "spiderweb-agent-template",
      env: {
        AGENT_ID: params.agentId,
        TASK_TYPE: params.taskType,
        AGENT_CONFIG: JSON.stringify(params.config)
      }
    });

    try {
      const result = await sandbox.runCommand({
        command: "node",
        args: ["-e", `require('./agent-runner').execute(${JSON.stringify(params.taskPayload)})`],
        timeout: 300000 // 5 minutes
      });

      return {
        success: true,
        output: result.stdout,
        error: result.stderr,
        duration: result.duration
      };
    } finally {
      await sandbox.close();
    }
  }
}

OpenClaw Bridge

Bridge Implementation

type OpenClawBridge struct {
    config     *OpenClawConfig
    wsConn     *websocket.Conn
    spiderweb  *SpiderwebClient
    reconnect  chan struct{}
    running    bool
    mutex      sync.RWMutex
}

func (b *OpenClawBridge) Connect() error {
    url := fmt.Sprintf("ws://%s:%d/ws", b.config.Host, b.config.Port)
    conn, _, err := websocket.DefaultDialer.Dial(url, nil)
    if err != nil {
        return fmt.Errorf("failed to connect to OpenClaw: %w", err)
    }
    
    b.wsConn = conn
    b.running = true
    
    go b.readMessages()
    go b.heartbeat()
    
    return nil
}

CI/CD Pipeline

name: Deploy Spiderweb System

on:
  push:
    branches: [ main, develop ]

jobs:
  deploy:
    runs-on: ubuntu-latest
    environment: ${{ github.ref == 'refs/heads/main' && 'production' || 'development' }}
    
    steps:
    - uses: actions/checkout@v3
    - name: Setup Node.js
      uses: actions/setup-node@v3
      with:
        node-version: '20'
        cache: 'npm'
    - name: Deploy to Trigger.dev
      run: npx trigger deploy
      env:
        TRIGGER_API_KEY: ${{ secrets.TRIGGER_API_KEY }}
        E2B_API_KEY: ${{ secrets.E2B_API_KEY }}
